  <script>
    // ====== Customize these ======
    const NAME = "Nirali"; // change to her name
    const YES_MESSAGE = "You just made me the happiest person ü•πüíó I can‚Äôt wait to see you. Valentine locked in üîíüå∏";

    // ====== Elements ======
    const theirName = document.getElementById("theirName");
    const yesBtn = document.getElementById("yesBtn");
    const noBtn = document.getElementById("noBtn");
    const arena = document.getElementById("arena");
    const hint = document.getElementById("hint");
    const modal = document.getElementById("modal");
    const msg = document.getElementById("msg");
    const closeBtn = document.getElementById("closeBtn");
    const hearts = document.getElementById("hearts");

    theirName.textContent = NAME;
    msg.textContent = YES_MESSAGE;

    // Put "No" somewhere nice at start
    placeNoRandom(true);

    // Move "No" away when hovered/touched
    let dodgeCount = 0;
    const dodgers = ["mouseover", "pointerdown", "touchstart"];
    dodgers.forEach(ev => noBtn.addEventListener(ev, (e) => {
      e.preventDefault();
      dodgeCount++;
      placeNoRandom(false);

      if (dodgeCount === 3) hint.textContent = "Stop trying to say noooo üò≠";
      if (dodgeCount === 6) hint.textContent = "Just press Yes already üòºüíó";
      if (dodgeCount >= 9) hint.textContent = "No button has ‚ú®teleport powers‚ú®";
    }, { passive:false }));

    // Yes opens modal
    yesBtn.addEventListener("click", () => {
      modal.classList.add("open");
      burstHearts(20);
    });

    closeBtn.addEventListener("click", () => modal.classList.remove("open"));
    modal.addEventListener("click", (e) => {
      if (e.target === modal) modal.classList.remove("open");
    });

    // ---- Helpers ----
    function placeNoRandom(isFirst){
      const r = arena.getBoundingClientRect();
      const b = noBtn.getBoundingClientRect();

      // Keep it inside the arena with padding
      const pad = 6;
      const maxX = r.width - b.width - pad;
      const maxY = r.height - b.height - pad;

      // If arena is too small, fallback
      const x = Math.max(pad, Math.random() * Math.max(pad, maxX));
      const y = Math.max(pad, Math.random() * Math.max(pad, maxY));

      noBtn.style.left = x + "px";
      noBtn.style.top  = y + "px";
      noBtn.style.transform = "none";

      if (!isFirst) {
        // tiny bounce effect
        noBtn.animate([
          { transform: "scale(1)" },
          { transform: "scale(1.06)" },
          { transform: "scale(1)" }
        ], { duration: 180, easing: "ease-out" });
      }
    }

    // Background floating hearts
    for (let i = 0; i < 18; i++) spawnFloatingHeart();

    function spawnFloatingHeart(){
      const el = document.createElement("div");
      el.className = "heart";
      el.textContent = ["üíó","üíñ","üíò","üíï","üíû"][Math.floor(Math.random()*5)];
      el.style.left = Math.random()*100 + "vw";
      el.style.bottom = (-10 - Math.random()*40) + "vh";
      el.style.fontSize = (14 + Math.random()*18) + "px";
      el.style.animationDuration = (6 + Math.random()*8) + "s";
      el.style.animationDelay = (Math.random()*4) + "s";
      hearts.appendChild(el);

      // recycle
      setTimeout(() => {
        el.remove();
        spawnFloatingHeart();
      }, 16000);
    }

    // Celebration burst when pressing Yes
    function burstHearts(n){
      for (let i = 0; i < n; i++){
        const el = document.createElement("div");
        el.className = "heart";
        el.textContent = ["üíó","üíñ","üíò","üíï","‚ú®"][Math.floor(Math.random()*5)];
        el.style.left = (40 + Math.random()*20) + "vw";
        el.style.bottom = "10vh";
        el.style.fontSize = (16 + Math.random()*20) + "px";
        el.style.animationDuration = (3 + Math.random()*2) + "s";
        el.style.animationDelay = "0s";
        hearts.appendChild(el);
        setTimeout(() => el.remove(), 4500);
      }
    }

    // Re-position No on resize
    window.addEventListener("resize", () => placeNoRandom(true));
